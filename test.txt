        <button type="button" id="addCertButton" class="btn btn-primary mt-3" onclick="addCertificateField()">Add Certificate</button>
        <button type="submit" class="btn btn-success mt-3">Submit All Certificates</button>
    </form>

<script>
let currentCertIndex = 0;
const certLabels = ['Dev Cert', 'QA Cert', 'UAT Cert', 'Prod Cert']; // Labels for the certificates

function addCertificateField() {
    const container = document.getElementById('inputContainer');
    const inputGroup = document.createElement('div');
    inputGroup.className = 'input-group';

    const label = document.createElement('label');
    label.textContent = certLabels[currentCertIndex] + ':';
    label.htmlFor = 'cert' + currentCertIndex;

    const input = document.createElement('input');
    input.type = 'text';
    input.className = 'form-control';
    input.id = 'cert' + currentCertIndex;
    input.placeholder = 'Enter ' + certLabels[currentCertIndex] + ' Details';

    inputGroup.appendChild(label);
    inputGroup.appendChild(input);
    container.appendChild(inputGroup);

    currentCertIndex++;
    if (currentCertIndex >= certLabels.length) {
        document.getElementById('addCertButton').disabled = true; // Disable button after all fields are added
    }
}

</script>
