import org.springframework.boot.actuate.endpoint.SanitizingFunction;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import java.util.Set;

@Configuration
public class ActuatorSanitizerConfig {

    @Bean
    public SanitizingFunction passwordSanitizer() {
        return (data, showUnsanitized) -> {
            Set<String> keysToSanitize = Set.of("password", "secret", "token", "spring.security.user.password");
            if (keysToSanitize.contains(data.getKey())) {
                return data.withValue("****");
            }
            return data;
        };
    }
}
