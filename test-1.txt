    <script>
        document.querySelector('.toggle-button').addEventListener('click', function() {
            const details = document.querySelector('.cert-details');
            details.style.display = details.style.display === 'none' ? 'block' : 'none';
            this.textContent = details.style.display === 'block' ? '-' : '+';
        });

        document.querySelectorAll('input[name="same_cert"]').forEach(input => {
            input.addEventListener('change', function() {
                const certCopyDiv = document.querySelector('.copy-cert-details');
                if (document.getElementById('yes').checked) {
                    document.getElementById('dev_copy').value = document.getElementById('dev').value;
                    document.getElementById('qa_copy').value = document.getElementById('qa').value;
                    document.getElementById('uat_copy').value = document.getElementById('uat').value;
                    document.getElementById('prod_copy').value = document.getElementById('prod').value;
                    certCopyDiv.style.display = 'block';
                } else {
                    certCopyDiv.style.display = 'none';
                }
            });
        });
